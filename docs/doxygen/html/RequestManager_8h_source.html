118 QByteArray makeRequest(QByteArray data)
119 {
120 int requestNumber;

122 QJsonDocument jsonDoc = QJsonDocument::fromJson(data, &parseError);
123
124 qDebug().noquote() << "--> Request received: " << jsonDoc.toJson(QJsonDocument::Indented);
125
126 QJsonObject jsonObjResponse;
127
128 QJsonObject jsonObjRequest = jsonDoc.object();
129
130 do
131 {
132 if (parseError.error == QJsonParseError::NoError)
133 {
134 requestNumber = jsonObjRequest.value("Request").toInt();
135 }
136 else
137 {
138 qCritical() << "JSON Parse Error: " << parseError.errorString();
139 jsonObjResponse = handleJsonParseError(parseError);
140 break;
141 }
142
143 if (requests_.contains(requestNumber))
144 {
145 //Request* request = requests_[requestNumber];
146 jsonObjResponse = requests_[requestNumber]->execute(jsonObjRequest, *Mutex_);
147 }
148 else
149 {
150 qCritical() << "Request not found";
151 qCritical() << "Invalid request: " << requestNumber;
152 }
153 } while (false);
154
155 QJsonDocument responseDoc(jsonObjResponse);
156 QByteArray responseByteArr = responseDoc.toJson();
157
158 return responseByteArr;
159 }
